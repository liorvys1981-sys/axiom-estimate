# k8s/deployments/api-gateway.yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  name: api-gateway
  namespace: axiom-production
  labels:
    app: api-gateway
    component: core
    version: "2.0.0"
spec:
  # GKE Autopilot escala automáticamente — no definimos replicas fijas
  replicas: 3
  selector:
    matchLabels:
      app: api-gateway
  strategy:
    type: RollingUpdate
    rollingUpdate:
      maxSurge: 1
      maxUnavailable: 0          # Zero-downtime para 100+ clientes
  template:
    metadata:
      labels:
        app: api-gateway
        component: core
      annotations:
        prometheus.io/scrape: "true"
        prometheus.io/port: "8000"
        prometheus.io/path: "/metrics"
    spec:
      serviceAccountName: axiom-service-account

      # Init container: verifica secrets antes de arrancar
      initContainers:
        - name: secrets-validator
          image: python:3.11-alpine
          command:
            - python3
            - -c
            - |
              import os, sys
              required = [
                'DATABASE_URL', 'JWT_SECRET_KEY',
                'STRIPE_SECRET_KEY', 'REDIS_URL'
              ]
              missing = [k for k in required if not os.environ.get(k)]
              if missing:
                print(f"FAIL: Missing secrets: {missing}")
                sys.exit(1)
              print("GREEN LIGHT: All secrets present")
          envFrom:
            - secretRef:
                name: axiom-runtime-secrets

      containers:
        - name: api-gateway
          image: us-east1-docker.pkg.dev/axiom-estimate-prod/axiom/api-gateway:latest
          imagePullPolicy: Always
          ports:
            - containerPort: 8000
              name: http
              protocol: TCP

          # Recursos optimizados para GKE Autopilot
          resources:
            requests:
              cpu: "250m"
              memory: "256Mi"
              ephemeral-storage: "1Gi"
            limits:
              cpu: "1000m"
              memory: "1Gi"
              ephemeral-storage: "2Gi"

          # Variables de entorno
          envFrom:
            - configMapRef:
                name: axiom-global-config
            - secretRef:
                name: axiom-runtime-secrets

          env:
            - name: SERVICE_NAME
              value: "api-gateway"
            - name: SERVICE_PORT
              value: "8000"
            - name: POD_NAME
              valueFrom:
                fieldRef:
                  fieldPath: metadata.name
            - name: POD_NAMESPACE
              valueFrom:
                fieldRef:
                  fieldPath: metadata.namespace
            - name: NODE_NAME
              valueFrom:
                fieldRef:
                  fieldPath: spec.nodeName

          # Health Checks — críticos para Self-Healing
          livenessProbe:
            httpGet:
              path: /health/live
              port: 8000
            initialDelaySeconds: 30
            periodSeconds: 10
            timeoutSeconds: 5
            failureThreshold: 3
            successThreshold: 1

          readinessProbe:
            httpGet:
              path: /health/ready
              port: 8000
            initialDelaySeconds: 15
            periodSeconds: 5
            timeoutSeconds: 3
            failureThreshold: 3
            successThreshold: 1

          startupProbe:
            httpGet:
              path: /health/startup
              port: 8000
            initialDelaySeconds: 10
            periodSeconds: 10
            failureThreshold: 30   # 5 minutos para arrancar

          # Lifecycle hooks
          lifecycle:
            preStop:
              exec:
                command: ["/bin/sh", "-c", "sleep 5"]

          # Security Context
          securityContext:
            allowPrivilegeEscalation: false
            runAsNonRoot: true
            runAsUser: 1000
            readOnlyRootFilesystem: true
            capabilities:
              drop: ["ALL"]

          volumeMounts:
            - name: tmp-dir
              mountPath: /tmp
            - name: app-logs
              mountPath: /app/logs

      volumes:
        - name: tmp-dir
          emptyDir: {}
        - name: app-logs
          emptyDir: {}

      # Tolerations para GKE Autopilot
      tolerations:
        - key: "kubernetes.io/arch"
          operator: "Equal"
          value: "amd64"
          effect: "NoSchedule"

      # Anti-affinity: distribuir pods en diferentes nodos
      affinity:
        podAntiAffinity:
          preferredDuringSchedulingIgnoredDuringExecution:
            - weight: 100
              podAffinityTerm:
                labelSelector:
                  matchExpressions:
                    - key: app
                      operator: In
                      values: ["api-gateway"]
                topologyKey: kubernetes.io/hostname

      # Termination grace period
      terminationGracePeriodSeconds: 30
