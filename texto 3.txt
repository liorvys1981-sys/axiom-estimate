# k8s/namespace.yaml
apiVersion: v1
kind: Namespace
metadata:
  name: axiom-production
  labels:
    app: axiom-estimate
    env: production
    managed-by: axiom-team
  annotations:
    scheduler.alpha.kubernetes.io/node-selector: ""
---
# ServiceAccount para los pods de AXIOM
apiVersion: v1
kind: ServiceAccount
metadata:
  name: axiom-service-account
  namespace: axiom-production
  annotations:
    # Workload Identity — acceso a Google Secret Manager sin credenciales
    iam.gke.io/gcp-service-account: axiom-gke-sa@axiom-estimate-prod.iam.gserviceaccount.com
---
# RBAC: Role para operaciones dentro del namespace
apiVersion: rbac.authorization.k8s.io/v1
kind: Role
metadata:
  name: axiom-pod-manager
  namespace: axiom-production
rules:
  - apiGroups: [""]
    resources: ["pods", "pods/log", "services", "endpoints"]
    verbs: ["get", "list", "watch", "create", "delete", "patch"]
  - apiGroups: ["apps"]
    resources: ["deployments", "replicasets"]
    verbs: ["get", "list", "watch", "update", "patch"]
  - apiGroups: [""]
    resources: ["configmaps"]
    verbs: ["get", "list"]
---
apiVersion: rbac.authorization.k8s.io/v1
kind: RoleBinding
metadata:
  name: axiom-pod-manager-binding
  namespace: axiom-production
subjects:
  - kind: ServiceAccount
    name: axiom-service-account
    namespace: axiom-production
roleRef:
  kind: Role
  apiRef: axiom-pod-manager
  apiGroup: rbac.authorization.k8s.io
---
# ResourceQuota — protección contra runaway costs
apiVersion: v1
kind: ResourceQuota
metadata:
  name: axiom-resource-quota
  namespace: axiom-production
spec:
  hard:
    requests.cpu: "20"
    requests.memory: "40Gi"
    limits.cpu: "40"
    limits.memory: "80Gi"
    pods: "100"
    services: "20"
    persistentvolumeclaims: "10"
---
# LimitRange — límites por defecto por container
apiVersion: v1
kind: LimitRange
metadata:
  name: axiom-limit-range
  namespace: axiom-production
spec:
  limits:
    - type: Container
      default:
        cpu: "500m"
        memory: "512Mi"
      defaultRequest:
        cpu: "100m"
        memory: "128Mi"
      max:
        cpu: "4"
        memory: "8Gi"
      min:
        cpu: "50m"
        memory: "64Mi"
