# k8s/deployments/office1-estimator.yaml
apiVersion: apps/v1
kind: Deployment
metadata:
  name: office1-estimator
  namespace: axiom-production
  labels:
    app: office1-estimator
    component: ai-vision
    office: "1"
    price-per-job: "9.00"
spec:
  replicas: 2
  selector:
    matchLabels:
      app: office1-estimator
  strategy:
    type: RollingUpdate
    rollingUpdate:
      maxSurge: 1
      maxUnavailable: 0
  template:
    metadata:
      labels:
        app: office1-estimator
        office: "1"
      annotations:
        prometheus.io/scrape: "true"
        prometheus.io/port: "8001"
    spec:
      serviceAccountName: axiom-service-account
      containers:
        - name: office1-estimator
          image: us-east1-docker.pkg.dev/axiom-estimate-prod/axiom/office1-estimator:latest
          imagePullPolicy: Always
          ports:
            - containerPort: 8001
              name: http

          # YOLOv8/SAM requiere más CPU/RAM
          resources:
            requests:
              cpu: "500m"
              memory: "1Gi"
              ephemeral-storage: "2Gi"
            limits:
              cpu: "2000m"
              memory: "4Gi"
              ephemeral-storage: "5Gi"

          envFrom:
            - configMapRef:
                name: axiom-global-config
            - secretRef:
                name: axiom-runtime-secrets

          env:
            - name: SERVICE_NAME
              value: "office1-ai-damage-estimator"
            - name: SERVICE_PORT
              value: "8001"
            - name: OFFICE_ID
              value: "1"
            - name: JOB_PRICE
              value: "9.00"
            - name: YOLO_MODEL_PATH
              value: "/app/models/yolov8n.pt"
            - name: SAM_MODEL_PATH
              value: "/app/models/sam_vit_h.pth"
            - name: CONFIDENCE_THRESHOLD
              value: "0.75"
            - name: MAX_CONCURRENT_JOBS
              value: "10"

          livenessProbe:
            httpGet:
              path: /health/live
              port: 8001
            initialDelaySeconds: 60    # YOLOv8 tarda en cargar
            periodSeconds: 15
            failureThreshold: 3

          readinessProbe:
            httpGet:
              path: /health/ready
              port: 8001
            initialDelaySeconds: 45
            periodSeconds: 10

          securityContext:
            allowPrivilegeEscalation: false
            runAsNonRoot: true
            runAsUser: 1000

          volumeMounts:
            - name: tmp-dir
              mountPath: /tmp
            - name: model-cache
              mountPath: /app/models

      # Volumen para cache de modelos AI
      volumes:
        - name: tmp-dir
          emptyDir: {}
        - name: model-cache
          emptyDir:
            sizeLimit: "10Gi"

      terminationGracePeriodSeconds: 60
