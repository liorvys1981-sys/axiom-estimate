# k8s/monitoring/axiom-monitoring.yaml
# ServiceMonitor para Prometheus
apiVersion: monitoring.coreos.com/v1
kind: ServiceMonitor
metadata:
  name: axiom-service-monitor
  namespace: axiom-production
  labels:
    app: axiom-estimate
    release: prometheus-stack
spec:
  selector:
    matchLabels:
      app: axiom-estimate
  namespaceSelector:
    matchNames:
      - axiom-production
  endpoints:
    - port: http
      path: /metrics
      interval: 30s
      scrapeTimeout: 10s
---
# PrometheusRule — Alertas críticas
apiVersion: monitoring.coreos.com/v1
kind: PrometheusRule
metadata:
  name: axiom-alerts
  namespace: axiom-production
  labels:
    release: prometheus-stack
spec:
  groups:
    - name: axiom.critical
      interval: 30s
      rules:
        # Pod caído
        - alert: AXIOMPodDown
          expr: |
            kube_deployment_status_replicas_available{
              namespace="axiom-production"
            } == 0
          for: 1m
          labels:
            severity: critical
            team: axiom
          annotations:
            summary: "CRÍTICO: Pod AXIOM caído"
            description: >
              El deployment {{ $labels.deployment }} tiene
              0 replicas disponibles. Self-healing activado.

        # CPU alta
        - alert: AXIOMHighCPU
          expr: |
            rate(container_cpu_usage_seconds_total{
              namespace="axiom-production"
            }[5m]) > 0.8
          for: 5m
          labels:
            severity: warning
          annotations:
            summary: "CPU alta en {{ $labels.pod }}"

        # Error rate alta
        - alert: AXIOMHighErrorRate
          expr: |
            rate(http_requests_total{
              namespace="axiom-production",
              status=~"5.."
            }[5m]) > 0.05
          for: 2m
          labels:
            severity: critical
          annotations:
            summary: "Error rate >5% en AXIOM"

        # Autonomía degradada
        - alert: AXIOMAutonomyDegraded
          expr: |
            axiom_autonomy_percentage < 95
          for: 5m
          labels:
            severity: critical
          annotations:
            summary: "Autonomía AXIOM por debajo del umbral"
            description: "Autonomía actual: {{ $value }}% (umbral: 99.9%)"
---
