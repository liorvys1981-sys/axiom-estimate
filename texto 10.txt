# k8s/ingress/axiom-ingress.yaml
apiVersion: networking.k8s.io/v1
kind: Ingress
metadata:
  name: axiom-main-ingress
  namespace: axiom-production
  annotations:
    # NGINX
    kubernetes.io/ingress.class: "nginx"
    nginx.ingress.kubernetes.io/ssl-redirect: "true"
    nginx.ingress.kubernetes.io/force-ssl-redirect: "true"
    nginx.ingress.kubernetes.io/proxy-body-size: "100m"
    nginx.ingress.kubernetes.io/proxy-read-timeout: "300"
    nginx.ingress.kubernetes.io/proxy-send-timeout: "300"
    nginx.ingress.kubernetes.io/proxy-connect-timeout: "60"

    # Rate Limiting — protección DDoS
    nginx.ingress.kubernetes.io/limit-rps: "100"
    nginx.ingress.kubernetes.io/limit-connections: "20"

    # Cert-Manager — TLS automático Let's Encrypt
    cert-manager.io/cluster-issuer: "letsencrypt-prod"

    # CORS
    nginx.ingress.kubernetes.io/enable-cors: "true"
    nginx.ingress.kubernetes.io/cors-allow-origin: "https://axiomestimate.com"
    nginx.ingress.kubernetes.io/cors-allow-methods: "GET,POST,PUT,DELETE,OPTIONS"

    # Security Headers
    nginx.ingress.kubernetes.io/configuration-snippet: |
      add_header X-Frame-Options "SAMEORIGIN" always;
      add_header X-Content-Type-Options "nosniff" always;
      add_header X-XSS-Protection "1; mode=block" always;
      add_header Referrer-Policy "strict-origin-when-cross-origin" always;
      add_header Strict-Transport-Security
        "max-age=31536000; includeSubDomains" always;
spec:
  tls:
    - hosts:
        - axiomestimate.com
        - admin.axiomestimate.com
        - api.axiomestimate.com
        - monitor.axiomestimate.com
      secretName: axiom-tls-secret

  rules:
    # API Gateway principal
    - host: api.axiomestimate.com
      http:
        paths:
          - path: /
            pathType: Prefix
            backend:
              service:
                name: api-gateway-service
                port:
                  number: 8000

    # Admin Panel
    - host: admin.axiomestimate.com
      http:
        paths:
          - path: /
            pathType: Prefix
            backend:
              service:
                name: api-gateway-service
                port:
                  number: 8000

    # Routing por path a cada Office
    - host: axiomestimate.com
      http:
        paths:
          - path: /api/v1/office1
            pathType: Prefix
            backend:
              service:
                name: office1-estimator-service
                port:
                  number: 8001
          - path: /api/v1/office2
            pathType: Prefix
            backend:
              service:
                name: office2-insurance-service
                port:
                  number: 8002
          - path: /api/v1/office3
            pathType: Prefix
            backend:
              service:
                name: office3-totalloss-service
                port:
                  number: 8003
          - path: /api/v1/office4
            pathType: Prefix
            backend:
              service:
                name: office4-lien-service
                port:
                  number: 8004
          - path: /api/v1/office5
            pathType: Prefix
            backend:
              service:
                name: office5-audit-service
                port:
                  number: 8005
          - path: /api/v1/office6
            pathType: Prefix
            backend:
              service:
                name: office6-cpo-service
                port:
                  number: 8006
          - path: /api/v1/office7
            pathType: Prefix
            backend:
              service:
                name: office7-gpu-service
                port:
                  number: 8007
          - path: /
            pathType: Prefix
            backend:
              service:
                name: api-gateway-service
                port:
                  number: 8000

# ClusterIssuer para Let's Encrypt
apiVersion: cert-manager.io/v1
kind: ClusterIssuer
metadata:
  name: letsencrypt-prod
spec:
  acme:
    server: https://acme-v02.api.letsencrypt.org/directory
    email: security@axiomestimate.com
    privateKeySecretRef:
      name: letsencrypt-prod-key
    solvers:
      - http01:
          ingress:
            class: nginx
